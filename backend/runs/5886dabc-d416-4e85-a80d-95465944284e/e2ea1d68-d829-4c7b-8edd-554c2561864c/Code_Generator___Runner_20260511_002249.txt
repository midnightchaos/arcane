def fibonacci(n):
    """
    Calculate the nth Fibonacci number using dynamic programming.
    
    Time complexity: O(n)
    Space complexity: O(n)
    """
    fib = [0, 1] + [0]*(n-1)  # Initialize the first two Fibonacci numbers and a list to store the rest
    for i in range(2, n):
        fib[i] = fib[i-1] + fib[i-2]
