def fibonacci(n):
    """
    Calculate the nth Fibonacci number.

    Time complexity: O(2^n)
    Space complexity: O(n)

    :param n: The position of the Fibonacci number to calculate (1-indexed)
    :return: The nth Fibonacci number
    """
    if n <= 0:
        raise ValueError("n must be a positive integer")
    elif n == 1:
        return 0
    elif n == 2:
        return 1

    a, b = 0, 1
    for i in range(2, n):
        a, b = b, a + b

    return b
