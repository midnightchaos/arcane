    """
    Calculate the nth Fibonacci number.

    :param n: The index of the Fibonacci number to calculate (0-based).
    :return: The nth Fibonacci number.
    """
    if n <= 1:

    a, b = 0, 1
    for i in range(2, n + 1):
        a, b = b, a + b
