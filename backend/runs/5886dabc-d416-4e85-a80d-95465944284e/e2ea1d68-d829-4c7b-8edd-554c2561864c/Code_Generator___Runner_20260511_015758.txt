def fibonacci(n):
    """
    Calculate the nth Fibonacci number using iterative method.

    Args:
        n (int): The index of the Fibonacci number to calculate (0-based).

    Returns:
        int: The nth Fibonacci number.
    """
    if n <= 1:
        return n

    a, b = 0, 1
    for i in range(2, n + 1):
        a, b = b, a + b

    return b
