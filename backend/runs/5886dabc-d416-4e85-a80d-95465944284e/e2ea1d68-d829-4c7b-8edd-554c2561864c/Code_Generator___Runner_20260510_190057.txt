def fibonacci(n):
    """
    Calculate the nth Fibonacci number.
    
    Parameters:
    n (int): The position of the Fibonacci number to be calculated.
    
    Returns:
    int: The nth Fibonacci number.
    """
    if n <= 0:
        return "Input should be a positive integer."
    elif n == 1:
        return 0
    elif n == 2:
        return 1
    else:
        fib_sequence = [0, 1]
        for i in range(2, n):
