def fibonacci(n):
    """
    Calculate the nth Fibonacci number.

    :param n: The position of the Fibonacci number to calculate.
    :return: The nth Fibonacci number.
    """
    if n <= 0:
        return None
    elif n == 1:
        return 0

        return 1
