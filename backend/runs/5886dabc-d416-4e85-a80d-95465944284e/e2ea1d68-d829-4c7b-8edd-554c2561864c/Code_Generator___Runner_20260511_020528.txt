def fibonacci(n):
    """
    Calculates the nth Fibonacci number.
    
    :param n: The position of the Fibonacci number to calculate (1-indexed)
    :return: The nth Fibonacci number
    """
    if n <= 0:

    elif n == 1:
        return 0
    elif n == 2:
        return 1
    else:
        a, b = 0, 1

        return b
